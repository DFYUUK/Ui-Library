local function Tween(instance, properties, duration, easingStyle, easingDirection)
	local tweenInfo = TweenInfo.new(duration or 0.3, easingStyle or Enum.EasingStyle.Quart, easingDirection or Enum.EasingDirection.Out)
	local tween = TweenService:Create(instance, tweenInfo, properties)
	tween:Play()
	return tween
end

				local function update(val, animate)
					value = math.clamp(math.floor(val + 0.5), min, max)
					local alpha = (max == min) and 0 or (value - min) / (max - min)
					local percent = math.floor(alpha * 100 + 0.5)
					ValueLabel.Text = tostring(value) .. " (" .. percent .. "%)"
					
					local duration = animate and 0.15 or 0.08
					local easing = Enum.EasingStyle.Quart
					
					Tween(Fill, {Size = UDim2.new(alpha, 0, 1, 0)}, duration, easing)
					Tween(Knob, {Position = UDim2.new(alpha, -7, 0.5, -7)}, duration, easing)
					
					callback(value)
				end
				
				local function pulseValueLabel()
					local originalColor = ValueLabel.TextColor3
					Tween(ValueLabel, {TextColor3 = Color3.fromRGB(0, 200, 170)}, 0.1)
					task.delay(0.2, function()
						Tween(ValueLabel, {TextColor3 = originalColor}, 0.3)
					end)
				end

						dragging = true
						
						-- Visual feedback
						Tween(Knob, {Size = UDim2.new(0, 18, 0, 18), Position = UDim2.new(Knob.Position.X.Scale, -9, 0.5, -9)}, 0.1)

						Tween(KnobStroke, {Color = Color3.fromRGB(0, 137, 123)}, 0.1)
						pulseValueLabel()
						
						-- Calculate position from input
						local pos = math.clamp((input.Position.X - Track.AbsolutePosition.X) / Track.AbsoluteSize.X, 0, 1)
						update(min + (max - min) * pos, false)
						
						local alpha = (max == min) and 0 or (value - min) / (max - min)
						local offsetY = -9
						Knob.Position = UDim2.new(alpha, offsetY, 0.5, offsetY)
					end
				end)

							local pos = math.clamp((input.Position.X - Track.AbsolutePosition.X) / Track.AbsoluteSize.X, 0, 1)

							GhostFill.Size = UDim2.new(ghostAlpha, 0, 1, 0)
							GhostFill.Visible = true
						end
						
						-- Mouse drag
						if dragging then
							local pos = math.clamp((input.Position.X - Track.AbsolutePosition.X) / Track.AbsoluteSize.X, 0, 1)
							update(min + (max - min) * pos, false)
							
							local alpha = (max == min) and 0 or (value - min) / (max - min)
							local offsetY = -9
							Knob.Position = UDim2.new(alpha, offsetY, 0.5, offsetY)

				end)

				UserInputService.InputEnded:Connect(function(input)
					if input.UserInputType == Enum.UserInputType.MouseButton1 or 
					   input.UserInputType == Enum.UserInputType.Touch then
						if dragging then
							dragging = false
							local normalSize = UDim2.new(0, 14, 0, 14)
							local offsetY = -7
							Tween(Knob, {Size = normalSize}, 0.15, Enum.EasingStyle.Back)
							Tween(KnobGlow, {BackgroundTransparency = 1}, 0.15)
							Tween(KnobStroke, {Color = Color3.fromRGB(200, 200, 200)}, 0.15)
						end
					end
				end)
				
				task.spawn(function()
					Fill.Size = UDim2.new(0, 0, 1, 0)
					local offsetY = -7
					Knob.Position = UDim2.new(0, offsetY, 0.5, offsetY)
					task.wait(0.1)
					update(default, true)
				end)
				
				return SectionObj
			end
